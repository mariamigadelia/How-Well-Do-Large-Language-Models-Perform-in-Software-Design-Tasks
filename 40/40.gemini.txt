title Academic Program Management System - Class Diagram

' --- Classes ---

class AcademicProgram {
    - programNumber: String
    - name: String
    + addCourse(course: Course): Void
    + removeCourse(course: Course): Void
}

class Course {
    - courseCode: String
    - courseName: String
    - startTime: Time
    - endTime: Time
    - daysOfWeek: List<String>
    - quotaLimit: Integer
    - weeklyHours: Integer
    + setQuotaLimit(limit: Integer): Void
    + setWeeklyHours(hours: Integer): Void
    + setDaysOfWeek(days: List<String>): Void
}

class Classroom {
    - classId: String
    - capacity: Integer
    - floor: Integer
    - block: String
}

class Instructor {
    - name: String
    - surname: String
    - title: String
    - specialty: String
    + lecture(): Void
    + mentor(student: Student): Void
    + adjustQuota(course: Course, newLimit: Integer): Void
    + modifyCourseSchedule(course: Course, hours: Integer, days: List<String>): Void
    + assignCourse(course: Course): Void
    + dropCourse(course: Course): Void
}

class Student {
    - studentNumber: String
    - name: String
    - major: String
    + enrollCourse(course: Course): Void
    + dropCourse(course: Course): Void
    + requestTranscript(): Transcript
    + requestDocument(docType: String): Document
    + submitSurvey(survey: Survey): Void
}

' --- Stub Helper Classes for Completeness ---
class Transcript { }
class Document { }
class Survey { }

' --- Relationships & Multiplicities ---

note top of Course
  <b>Schedule Attributes:</b>
  Encapsulates start/end times, 
  weekly hour allocations, and 
  active days (e.g., ["Monday", "Wednesday"]).

note bottom of Instructor
