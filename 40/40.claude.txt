@startuml AcademicProgramDiagram

class AcademicProgram {
    - programNumber : int
    - name : String
    --
    + addCourse(c : Course) : void
    + removeCourse(c : Course) : void
}

class Course {
    - courseName : String
    - courseCode : String
    - startTime : Time
    - endTime : Time
    - days : String[]
    - quotaLimit : int
    --
    + getSchedule() : String
}

class Classroom {
    - classId : String
    - capacity : int
    - floor : int
    - block : String
}

class Instructor {
    - name : String
    - surname : String
    - title : String
    - specialty : String
    --
    + lecture(s : Student) : void
    + mentor(s : Student) : void
    + increaseQuota(c : Course) : void
    + decreaseQuota(c : Course) : void
    + changeWeeklyHours(c : Course) : void
    + changeCourseDays(c : Course) : void
    + addCourse(c : Course) : void
    + dropCourse(c : Course) : void
}

class Student {
    - studentNumber : int
    - name : String
    - major : String
    --
    + enroll(c : Course) : void
    + dropCourse(c : Course) : void
    + receiveTranscript() : Transcript
    + requestDocument() : Document
    + conductSurvey() : void
}
