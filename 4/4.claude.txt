' ================= CLASSES =================

class Consortium {
  - name : String
  - sharedSystemCost : double
  + apportionCost() : void
  + clearTransaction(t : Transaction) : boolean
}

class CentralComputer {
  - location : String
  + routeTransaction(t : RemoteTransaction) : void
  + verifyCard(card : CashCard) : boolean
}

class Bank {
  - bankCode : String
  - name : String
  + processTransaction(t : Transaction) : boolean
  + countCashCardCustomers() : int
}

class BankComputer {
  - address : String
  + openAccount(c : Customer) : Account
  + updateAccount(a : Account, t : Transaction) : void
  + lockAccount(a : Account) : void
  + unlockAccount(a : Account) : void
}

class Account {
  - accountNumber : String
  - balance : double
  - creditLimit : double
  - type : String
  - lockOwner : Transaction
  + post(amount : double) : void
  + getBalance() : double
  + acquireLock(t : Transaction) : boolean
  + releaseLock(t : Transaction) : void
}

class Customer {
  - name : String
  - address : String
  - customerId : String
}

class CashCard {
  - cardNumber : String
  - serialNumber : String
  - expiryDate : Date
  + getBank() : Bank
}

class CashierStation {
  - stationId : String
  + printReceipt() : Receipt
  + submitTransaction(t : CashierTransaction) : void
}

class Cashier {
  - employeeId : String
  - name : String
  + enterAccountData() : void
  + enterTransactionData() : void
}

class ATM {
  - machineId : String
  - location : String
  - cashOnHand : double
  + acceptCard(card : CashCard) : boolean
  + interactWithUser() : void
  + dispenseCash(amount : double) : void
  + printReceipt() : Receipt
  + ejectCard() : void
}

abstract class Transaction {
  - transactionId : String
  - dateTime : DateTime
  - amount : double
  - type : TransactionKind
  + execute() : boolean
  + log() : void
}

class CashierTransaction extends Transaction {
  + enteredBy() : Cashier
}

class RemoteTransaction extends Transaction {
  + getStartingATM() : ATM
}

class Receipt {
  - receiptNumber : String
  - dateTime : DateTime
  - balanceAfter : double
  + print() : void
}

enum TransactionKind {
  WITHDRAWAL
  DEPOSIT
  TRANSFER
  BALANCE_INQUIRY
}
