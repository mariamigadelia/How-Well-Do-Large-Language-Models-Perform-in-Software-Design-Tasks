class Farm {
  - farmId: int
  - name: String
  - address: String

}

class Certification {
  - certificationNumber: int
  - date: Date
  - description: String
  - validity: String
  + assessEligibility()
  + conductInspection()
  + issueCertificate()
}

class Inspection {
  - inspectionId: int
  - inspectionDate: Date
  - inspectorName: String
  - result: String
  + performInspection()
}

class Monitoring {
  - monitoringId: int
  - activityDate: Date
  - description: String
  - outcome: String
  + trackFarmingActivities()
  + verifyCompliance()
  + storeResults()
}

class OrganicStandard {
  - standardId: int
  - standardName: String
  - requirements: String
}

class CertificationRecord {
  - recordNumber: int
  - renewalDate: Date
  - status: String
  + manageRenewal()
  + updateStatus()
}

class Certificate {
  - certificateId: int
  - issueDate: Date
  - expiryDate: Date
  - certificateDetails: String
}
