title Supermarket System Class Diagram

' ==================== ENUMERATIONS ====================
enum ProductType {
  ALIMENTATION
  PASTRY_BAKERY
  BUTCHER
}

enum StorageLocation {
  SHELF
  COLD_CHAMBER
  FREEZER
}

enum CustomerType {
  NORMAL
  EXTRA
}

enum OrderStatus {
  PENDING
  PREPARED
  VERIFIED
  COMPLETED
}

enum DeliveryStatus {
  IN_PROGRESS
  COMPLETED
}

enum PaymentMethod {
  CASH
  CARD
}

' ==================== PRODUCT ====================
class Product {
  - productID : String
  - name : String
  - type : ProductType
  - expiryDate : Date
  - price : double
  - storageLocation : StorageLocation
  + checkExpiry() : boolean
  + getPrice() : double
}

class ProductGroup {
  - groupID : String
  - continent : String
  + organizeProducts() : void
}

' ==================== STOCK / INVENTORY ====================
class StockForm {
  - formID : String
  - product : Product
  - quantity : int
  + getQuantity() : int
}

class InventorySystem {
  - forms : List<StockForm>
  + recordProduct(p : Product, qty : int) : void
  + verifyCount(p : Product) : boolean
}

' ==================== PEOPLE ====================
abstract class Customer {
  - customerID : String
  - name : String
  - type : CustomerType
  + chooseProducts() : void
  + checkProductInfo(p : Product) : void
  + proceedToCashier() : void
}

class NormalCustomer
class ExtraCustomer

class Employee {
  - employeeID : String
  - name : String
  + orientCustomer(c : Customer) : void
  + placeProductOnShelf(p : Product) : void
}

class Cashier {
  - cashierID : String
  + scanProduct(p : Product, qty : int) : void
  + processPayment(pay : Payment) : void
  + provideInvoice(o : Order) : Invoice
}

class ChiefOfStore {
  - chiefID : String
  + checkProductCount() : void
}

class ChiefOfOrder {
  - chiefID : String
  + sendOrder(o : Order) : void
  + verifyOrder(o : Order, pi : ProformaInvoice) : boolean
}

class BakeryPastryAgent {
  - agentID : String
  + prepareGoods(o : Order) : void
  + packageGoods(o : Order) : void
}

class Controller {
  - controllerID : String
  + checkDelivery(d : Delivery) : void
}

class GeneralDirector {
  - directorID : String
}

' ==================== SHOPPING ====================
class Basket {
  - basketID : String
  + addProduct(p : Product) : void
  + removeProduct(p : Product) : void
}

' ==================== PAYMENT ====================
abstract class Payment {
  - paymentID : String
  - amount : double
  - method : PaymentMethod
  + process() : void
}

class CashPayment {
  - cashGiven : double
  + giveCash() : void
}

class CardPayment {
  - cardNumber : String
  + presentCard() : void
}

class Invoice {
  - invoiceID : String
  - totalAmount : double
  - date : Date
}

' ==================== ORDER & DELIVERY ====================
class Order {
  - orderID : String
  - products : List<Product>
  - quantities : Map<Product,int>
  - status : OrderStatus
}

class Delivery {
  - deliveryID : String
  - order : Order
  - status : DeliveryStatus
  + startDelivery() : void
  + completeDelivery() : void
}

class ProformaInvoice {
  - proformaID : String
  - stump : String
  - original : String
  + transferToDirector() : void
}

' ==================== INHERITANCE ====================
Customer <|-- NormalCustomer
Customer <|-- ExtraCustomer
Payment <|-- CashPayment
Payment <|-- CardPayment

' ==================== ASSOCIATIONS ====================
ProductGroup "1" o-- "*" Product : organizes >
InventorySystem "1" o-- "*" StockForm : maintains >
StockForm "1" --> "1" Product : records >
ChiefOfStore "1" ..> "*" StockForm : counts >

Customer "1" --> "1" Basket : uses >
Basket "1" o-- "*" Product : contains >

Cashier "1" --> "*" Payment : processes >
Cashier "1" --> "*" Invoice : provides >
Customer "1" --> "*" Payment : makes >
Payment "1" --> "0..1" Invoice : generates >

Employee "1" ..> "*" Customer : orients >
Employee "1" ..> "*" Product : placesOnShelf >

Order "1" o-- "*" Product : includes >
ChiefOfOrder "1" --> "*" Order : sends >
BakeryPastryAgent "1" ..> "*" Order : prepares >
ChiefOfOrder "1" ..> "*" ProformaInvoice : verifiesAgainst >

Delivery "1" --> "1" Order : delivers >
Controller "1" ..> "*" Delivery : checks >
ProformaInvoice "1" --> "1" GeneralDirector : transferredTo >

@enduml
