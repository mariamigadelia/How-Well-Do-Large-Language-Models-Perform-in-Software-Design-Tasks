class FlightReservationSystem {
  + checkConnectionFeasibility()
  + checkSchedule()
  + makeReservation()
  + issueTicket()
  + showFlightInformation()
}

class Customer {
  - customerId: int
  - title: String
  - name: String
  - address: String
  + makeReservation()
}

class Flight {
  - flightNumber: String
  - flightDate: Date
  - departureTime: Time
  - arrivalTime: Time
  - status: String
  + checkSchedule()
}

class Plane {
  - planeId: int
  - model: String
  - capacity: int
}

class Seat {
  - seatNumber: String
  - seatClass: String
  - isAvailable: boolean
}

class Reservation {
  - reservationId: int
  - reservationDate: Date
  - status: String
  + reserveSeat()
  + cancelReservation()
}

class Ticket {
  - ticketNumber: int
  - issueDate: Date
  - ticketDetails: String
}

class Airport {
  - airportCode: String
  - airportName: String
  - city: String
  - country: String
}

class Route {
  - routeId: int
  + validateRoute()
}

class Span {
  - spanId: int
  - startTime: Time
  - terminationTime: Time
}

class Airline {
  - airlineId: int
  - airlineName: String
}

class BasicAirline {
  - basicAirlineId: int
  - distance: double
  - duration: double
}

class ConnectingFlight {
  - connectionId: int
  - connectionTime: Time
  + checkFeasibility()
}

class InformationBoard {
  - boardId: int
  - location: String
  + displayFlightInfo()
}

class FacilityInformation {
  - facilityId: int
  - facilityType: String
  - description: String
  - nearbyCityDetails: String
}
