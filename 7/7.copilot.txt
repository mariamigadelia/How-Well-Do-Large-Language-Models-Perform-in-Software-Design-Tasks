class TimberFramedHouse {
    -numLogs : int
    -numBricks : int
    -numNails : int
    -rooms : int
    -floors : int
    +getDetails() : String
}

class WoodenLog {
    -logId : String
}

class AdobeBrick {
    -brickId : String
}

class Nail {
    -nailId : String
}

class Carpenter {
    -name : String
    -hourlyWage : float
    +workOnHouse(house : TimberFramedHouse)
}

class Hammer {
    -hammerId : String
}

class Saw {
    -sawId : String
}
