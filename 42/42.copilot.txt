' ==================== ENUMERATIONS ====================
enum CommodityStatus {
  AVAILABLE
  IN_TRANSIT
  EXPIRED
}

enum OrderStatus {
  PENDING
  CONFIRMED
  SHIPPED
  COMPLETED
}

enum ShipmentStatus {
  PENDING
  IN_PROGRESS
  DELIVERED
}

enum DocumentType {
  INVOICE
  CERTIFICATE
  LICENSE
  SHIPPING_DOC
}

' ==================== COMPANY ====================
class Company {
  - companyID : String
  - name : String
  - address : String
  - contactInfo : String
  - certifications : String
  - exportLicenses : String
  + manageCommodities()
  + manageOrders()
  + manageShipments()
  + manageDocuments()
}

' ==================== COMMODITY ====================
class Commodity {
  - commodityID : String
  - name : String
  - type : String
  - description : String
  - status : CommodityStatus
  - qualityStandards : String
  - shelfLife : int
}

' ==================== ORDER ====================
class Order {
  - orderID : String
  - date : Date
  - quantity : int
  - destination : String
  - status : OrderStatus
}

' ==================== SHIPMENT ====================
class Shipment {
  - shipmentNumber : String
  - date : Date
  - route : String
  - status : ShipmentStatus
}

' ==================== DOCUMENT ====================
class Document {
  - documentID : String
  - date : Date
  - type : DocumentType
  - details : String
}
