class WatCardSystem {
  + grantAccess()
  + processOnlineDeposit()
  + processPayment()
  + recordUsage()
}

class Student {
  - studentNumber: int
  - name: String
  - email: String
  - telephoneNumber: String
  - major: String
  - cardValidity: String
}

class WatCard {
  - cardId: int
  - cardNumber: String
  - issueDate: Date
  - expiryDate: Date
  - status: String
  + validateCard()
}

class DebitAccount {
  - accountId: int
  - balance: double
  + depositMoney()
  + withdrawMoney()
}

class OnlineDeposit {
  - depositId: int
  - amount: double
  - depositDate: Date
  - paymentMethod: String
  + makeDeposit()
}

class UniversityFee {
  - feeId: int
  - feeType: String
  - amount: double
  - description: String
}

class Payment {
  - paymentId: int
  - paymentDate: Date
  - amount: double
  + processPayment()
}

class SecureArea {
  - areaId: int
  - areaName: String
  - location: String
  - accessLevel: String
}

class AccessRecord {
  - accessId: int
  - accessTime: DateTime
  - accessStatus: String
  + recordAccess()
}

class UsageRecord {
  - usageId: int
  - usageDateTime: DateTime
  - usageType: String
  - description: String
  + createUsageRecord()
}
