  - paymentId: int
  - date: Date
  - purpose: String

  + issueTicket()
}

  - attendanceId: int
  - timestamp: DateTime
  - sessionDetails: String
  + recordCheckIn()
}

  - conferenceId: int
  - conferenceName: String
  - location: String
  - startDate: Date
  - endDate: Date
}

class Session {
  - sessionId: int
  - sessionTitle: String
  - sessionDetails: String

  - endTime: DateTime
}
