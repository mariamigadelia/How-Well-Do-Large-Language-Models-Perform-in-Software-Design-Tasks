@startuml Travel Expense Management System

skinparam classAttributeIconSize 0
skinparam class {
    BackgroundColor #F5F5F5
    BorderColor #333333
    ArrowColor #333333
}

class Employee {
    - employeeID : String
    - name : String
    - email : String
    - phoneNumber : String
    - department : String
    - travelExpenseHistory : List<ExpenseReport>
    --
    + submitExpenseReport(report : ExpenseReport) : void
    + viewExpenseHistory() : List<ExpenseReport>
    + updateContactInfo(email : String, phone : String) : void
}

class ContactInformation {
    - email : String
    - phoneNumber : String
    - address : String
    --
    + getContactDetails() : String
    + updateContact() : void
}

class ExpenseReport {
    - reportID : String
    - expenseCategory : ExpenseCategory
    - description : String
    - amount : double
    - dateIncurred : Date
    - receipt : Receipt
    - status : ReportStatus
    --
    + createReport() : void
    + addReceipt(receipt : Receipt) : void
    + calculateTotal() : double
    + submitForApproval() : void
}

class Receipt {
    - receiptID : String
    - fileImage : Blob
    - vendor : String
    - amount : double
    - dateOfPurchase : Date
    --
    + uploadReceipt() : void
    + validateReceipt() : boolean
}

class Approval {
    - approvalCode : String
    - approvalStatus : ApprovalStatus
    - approverID : String
    - approvalDate : Date
    - comments : String
    --
    + approve() : void
    + reject(reason : String) : void
    + checkCompliance() : boolean
}

class Reimbursement {
    - reimbursementID : String
    - reimbursementAmount : double
    - processingDate : Date
    - paymentMethod : String
    - status : ReimbursementStatus
    --
    + processReimbursement() : void
    + summarizeApprovedExpenses() : double
    + generatePaymentRecord() : void
}

class ExpenseAnalytics {
    - analyticsID : String
    - reportingPeriod : String
    - totalSpending : double
    --
    + trackBudget(department : String) : double
    + analyzeByCategory() : Map<ExpenseCategory, Double>
    + generateSpendingReport() : Report
    + checkPolicyCompliance() : boolean
}

class Budget {
    - budgetID : String
    - department : String
    - allocatedAmount : double
    - spentAmount : double
    - remainingAmount : double
    --
    + calculateRemaining() : double
    + isOverBudget() : boolean
}

enum ExpenseCategory {
    AIRFARE
    LODGING
    MEALS
    TRANSPORTATION
    ENTERTAINMENT
    MISCELLANEOUS
}

enum ApprovalStatus {
    PENDING
    APPROVED
    REJECTED
}

enum ReportStatus {
    DRAFT
    SUBMITTED
    UNDER_REVIEW
    APPROVED
    REIMBURSED
}

enum ReimbursementStatus {
    PENDING
    PROCESSING
    COMPLETED
    FAILED
}

' Relationships
Employee "1" *-- "1" ContactInformation : has >
Employee "1" --> "0..*" ExpenseReport : submits >
ExpenseReport "1" *-- "1..*" Receipt : contains >
ExpenseReport "1" --> "1" Approval : requires >
Approval "1" --> "0..1" Reimbursement : triggers >
Reimbursement "0..*" --> "1" Employee : reimburses >

ExpenseReport ..> ExpenseCategory : uses
Approval ..> ApprovalStatus : uses
ExpenseReport ..> ReportStatus : uses
Reimbursement ..> ReimbursementStatus : uses

ExpenseAnalytics "1" --> "0..*" ExpenseReport : analyzes >
ExpenseAnalytics "1" --> "1..*" Budget : tracks >
Budget "1" --> "1" Employee : associated with >

@enduml
